## Usage

```jsx
import React from 'react';

import DatePicker from '@neoaren/date-picker';

const App = () => {

	const handler = date => console.log(date);

	return (
		<DatePicker id="my-datepicker" defaultValue={new Date()} onChange={handler} />
	);

};
```
